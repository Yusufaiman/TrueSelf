## PART 1: EXPANDED IDENTITY SYSTEM (15 Types)

### **Tier 1: GROUNDED IDENTITIES (High Stability + High Clarity)**

#### 1. **The Anchored**

- **Core Pattern**: Stable, self-aware, authentic within chosen life framework
- **Traits**: Consistent identity across contexts, knows values, lives aligned
- **Strengths**: Stability, authenticity, resilience, purposeful living
- **Weaknesses**: May resist growth, rigid in beliefs, inflexible to change
- **Key Dimensions**: `identityStability` (90%), `authenticity` (85%), `selfAwareness` (85%), `innerConsistency` (90%)
- **Opposite**: The Shifter

#### 2. **The Rooted Explorer**

- **Core Pattern**: Stable core identity + curiosity/growth mindset
- **Traits**: Knows who they are but explores possibilities; change doesn't destabilize them
- **Strengths**: Adaptability within clarity, growth-oriented, resilient
- **Weaknesses**: May overcommit to exploration, dilute focus
- **Key Dimensions**: `identityStability` (75%), `decisionClarity` (80%), `flexibility` (75%), `selfAwareness` (80%)
- **New Dimension Needed**: `growthMindset`

#### 3. **The Idealist**

- **Core Pattern**: Driven by values and vision; identity shaped by purpose
- **Traits**: Strong internal compass, principled, future-oriented
- **Strengths**: Purpose-driven, inspiring, principled action
- **Weaknesses**: Perfectionism, disappointment when reality doesn't match vision, rigid values
- **Key Dimensions**: `decisionClarity` (85%), `authenticity` (80%), `emotionalAlignment` (75%), `socialExpression` (70%)
- **Note**: Maps to "cold visionary" from main personality types

## PART 4: NEW SCORING SYSTEM LOGIC

### **Step 1: Calculate Raw Dimension Scores**

For each dimension:

1. Sum responses for questions belonging to that dimension
2. Reverse-score reverse-coded questions (1→5, 2→4, 3→3, 4→2, 5→1)
3. Normalize to 0-100 scale: `(sum / max_possible) * 100`

**Example:**

```
selfAwareness questions: Q1-Q6
Response: [5, 4, 5, 2, 4, 1]
Raw sum: 21
Reverse Q2, Q4, Q6: [5, 4-(2), 5, 5-(2), 4, 5-(1)]
Adjusted: [5, 4, 5, 3, 4, 4]
Sum: 25
Max possible (6 questions × 5): 30
Score: (25/30) × 100 = 83.3
```

### **Step 2: Apply Weighted Normalization**

```javascript
function normalizeScores(rawScores) {
  const weights = {
    selfAwareness: 1.2,
    authenticity: 1.2,
    externalInfluence: 0.8, // Inverse: lower is better
    identityStability: 1.1,
    emotionalAlignment: 1.0,
    decisionClarity: 1.1,
    innerConsistency: 1.0,
    socialExpression: 0.9,
  };

  const weightedTotal = Object.keys(rawScores).reduce((sum, dim) => {
    return sum + rawScores[dim] * weights[dim];
  }, 0);

  const maxWeightedTotal = 100 * Object.values(weights).reduce((a, b) => a + b);

  return {
    ...rawScores,
    weightedAverage: (weightedTotal / maxWeightedTotal) * 100,
  };
}
```

### **Step 4: Profile Matching (Core Algorithm)**

```javascript
function calculateSimilarityScore(userScores, identityProfile) {
  const dimensions = Object.keys(userScores);
  let totalDistance = 0;

  dimensions.forEach((dim) => {
    const userScore = userScores[dim];
    const profileScore = identityProfile[dim];
    const distance = Math.abs(userScore - profileScore);
    totalDistance += distance;
  });

  const averageDistance = totalDistance / dimensions.length;
  const tolerance = identityProfile.tolerance || 15;

  // Similarity: 0-100 (100 = perfect match)
  // If within tolerance, higher score
  if (averageDistance <= tolerance) {
    return 100 - averageDistance * (50 / tolerance); // Max penalty: 50
  } else {
    return Math.max(0, 50 - averageDistance * 5); // Steep penalty outside tolerance
  }
}

function identifyType(userScores) {
  const matches = {};

  Object.keys(identityProfiles).forEach((typeKey) => {
    const profile = identityProfiles[typeKey];
    matches[typeKey] = calculateSimilarityScore(userScores, profile);
  });

  // Sort by similarity
  const ranked = Object.entries(matches)
    .sort(([, scoreA], [, scoreB]) => scoreB - scoreA)
    .slice(0, 3); // Top 3 matches

  return {
    primary: ranked[0][0],
    primaryScore: ranked[0][1],
    secondary: ranked[1][0],
    secondaryScore: ranked[1][1],
    tertiary: ranked[2][0],
    tertiaryScore: ranked[2][1],
    allMatches: matches,
  };
}
```

### **Step 5: Anti-Bias Safeguards**

```javascript
function applyBiasPrevention(userScores, result) {
  // PREVENT "THE BECOMING" FROM BEING DEFAULT

  // The Becoming should ONLY match if:
  // 1. selfAwareness >= 60 (must be aware of self)
  // 2. Some dimension changed significantly (growth signal)
  // 3. innerConsistency between 45-65 (in process, not total chaos)

  if (result.primary === "the-becoming") {
    if (userScores.selfAwareness < 60) {
      // User is not self-aware enough for "Becoming"
      // Reassign to more appropriate type
      return reassignLowestSimilarity(result, 0.15);
    }
    if (userScores.innerConsistency < 45 || userScores.innerConsistency > 65) {
      // Not in active becoming state
      return reassignLowestSimilarity(result, 0.1);
    }
  }

  return result;
}

function reassignLowestSimilarity(result, threshold) {
  // If secondary type's score is within threshold of primary, use secondary
  if (result.primaryScore - result.secondaryScore < threshold * 100) {
    return {
      ...result,
      primary: result.secondary,
      primaryScore: result.secondaryScore,
      secondary: result.primary,
      secondaryScore: result.primaryScore,
    };
  }

  return result;
}
```

  15,
  18, // externalInfluence
  20,

  38,

  44,
  46,
  48, // socialExpression
];

function scoreDimensions(answers) {
  // answers: array of 48 values (1-5)
  const scores = {};

  Object.keys(dimensionQuestions).forEach((dimension) => {
    const questionIndices = dimensionQuestions[dimension];
    let rawSum = 0;

    questionIndices.forEach((qIndex) => {
      let response = answers[qIndex - 1]; // 0-indexed

      // Reverse-score if needed
      if (reverseScored.includes(qIndex)) {
        response = 6 - response; // 1→5, 2→4, etc.
      }

      rawSum += response;
    });

    const maxPossible = questionIndices.length * 5;
    scores[dimension] = (rawSum / maxPossible) * 100;
  });

  return scores;
}

function identifyPersonalityType(answers) {
  // Step 1: Score dimensions
  const userScores = scoreDimensions(answers);

  // Step 2: Calculate similarity to all profiles
  const matches = {};
  Object.keys(identityProfiles).forEach((typeKey) => {
    const profile = identityProfiles[typeKey];
    matches[typeKey] = calculateSimilarityScore(userScores, profile);
  });

  // Step 3: Rank matches
  const ranked = Object.entries(matches).sort(
    ([, scoreA], [, scoreB]) => scoreB - scoreA,
  );

  let result = {
    primary: ranked[0][0],
    primaryScore: ranked[0][1],
    secondary: ranked[1][0],
    secondaryScore: ranked[1][1],
    tertiary: ranked[2][0],
    tertiaryScore: ranked[2][1],
    allMatches: matches,
    dimensionScores: userScores,
  };

  // Step 4: Apply bias prevention
  result = applyBiasPrevention(userScores, result);

  return result;
}
